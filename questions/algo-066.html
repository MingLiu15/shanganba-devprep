线段树（Segment Tree）

算法题 困难
标签：线段树, 区间查询, 区间更新, 数据结构

📑 题目

什么是线段树？如何实现区间查询和区间更新？

💡 参考答案

**线段树定义：**
一种二叉树数据结构，用于处理区间查询和区间更新问题。每个节点代表一个区间，存储该区间的聚合信息（和、最大值、最小值等）。

**特点：**
- 查询时间：O(log n)
- 更新时间：O(log n)
- 空间：O(4n)

**应用场景：**
- 区间求和、区间最值
- 区间修改（单点/区间）
- 逆序对统计
- 扫描线算法

**Lazy Propagation（懒标记）：**
- 区间更新时延迟下传标记
- 减少不必要的更新操作
- 真正实现O(log n)区间更新

💻 代码示例

class SegmentTree: def __init__(self, nums): self.n = len(nums) self.tree = [0] * (4 * self.n) self.lazy = [0] * (4 * self.n) self.nums = nums self._build(0, 0, self.n - 1) def _build(self, node, start, end): if start == end: self.tree[node] = self.nums[start] else: mid = (start + end) // 2 self._build(2 * node + 1, start, mid) self._build(2 * node + 2, mid + 1, end) self.tree[node] = self.tree[2 * node + 1] + self.tree[2 * node + 2] def _push_down(self, node, start, end): if self.lazy[node] != 0: mid = (start + end) // 2 self.tree[2 * node + 1] += self.lazy[node] * (mid - start + 1) self.lazy[2 * node + 1] += self.lazy[node] self.tree[2 * node + 2] += self.lazy[node] * (end - mid) self.lazy[2 * node + 2] += self.lazy[node] self.lazy[node] = 0 def update_range(self, left, right, val): self._update_range(0, 0, self.n - 1, left, right, val) def _update_range(self, node, start, end, left, right, val): if right < start or end < left: return if left <= start and end <= right: self.tree[node] += val * (end - start + 1) self.lazy[node] += val return self._push_down(node, start, end) mid = (start + end) // 2 self._update_range(2 * node + 1, start, mid, left, right, val) self._update_range(2 * node + 2, mid + 1, end, left, right, val) self.tree[node] = self.tree[2 * node + 1] + self.tree[2 * node + 2] def query_range(self, left, right): return self._query_range(0, 0, self.n - 1, left, right) def _query_range(self, node, start, end, left, right): if right < start or end < left: return 0 if left <= start and end <= right: return self.tree[node] self._push_down(node, start, end) mid = (start + end) // 2 left_sum = self._query_range(2 * node + 1, start, mid, left, right) right_sum = self._query_range(2 * node + 2, mid + 1, end, left, right) return left_sum + right_sum # 使用示例 nums = [1, 3, 5, 7, 9, 11] st = SegmentTree(nums) print(st.query_range(1, 3)) # 3+5+7=15 st.update_range(1, 3, 2) print(st.query_range(1, 3)) # 5+7+9=21
