接口自动化测试框架

自动化测试 困难
标签：接口测试, requests, pytest, 框架设计

📑 题目

如何设计接口自动化测试框架？

💡 参考答案

**框架组成：**
1. 配置管理：环境、URL、认证
2. 请求封装：统一请求处理
3. 数据驱动：YAML/JSON测试数据
4. 断言封装：响应验证
5. 报告生成：Allure报告

**目录结构：**
```
api_test/
├── config/ # 配置文件
├── api/ # 接口封装
├── testcases/ # 测试用例
├── data/ # 测试数据
├── utils/ # 工具函数
└── conftest.py # 共享fixture
```

💻 代码示例

# config/config.py import yaml class Config: def __init__(self, env="test"): with open(f"config/{env}.yaml") as f: self.config = yaml.safe_load(f) @property def base_url(self): return self.config["base_url"] # api/base.py import requests class BaseAPI: def __init__(self, base_url): self.base_url = base_url self.session = requests.Session() def request(self, method, endpoint, **kwargs): url = f"{self.base_url}{endpoint}" response = self.session.request(method, url, **kwargs) return response def get(self, endpoint, **kwargs): return self.request("GET", endpoint, **kwargs) def post(self, endpoint, **kwargs): return self.request("POST", endpoint, **kwargs) # api/user_api.py class UserAPI(BaseAPI): def login(self, username, password): return self.post("/api/login", json={ "username": username, "password": password }) def get_user(self, user_id): return self.get(f"/api/users/{user_id}") # conftest.py import pytest from config.config import Config from api.user_api import UserAPI @pytest.fixture(scope="session") def config(): return Config() @pytest.fixture(scope="session") def user_api(config): return UserAPI(config.base_url) # testcases/test_user.py import pytest import allure @allure.feature("用户接口") class TestUserAPI: @allure.story("登录") def test_login_success(self, user_api): response = user_api.login("admin", "password") assert response.status_code == 200 assert response.json()["token"] is not None @allure.story("获取用户") def test_get_user(self, user_api): response = user_api.get_user(1) assert response.status_code == 200 assert response.json()["id"] == 1
