快速排序算法实现

C/C++ 中等
标签：快速排序, 算法, 分治, 递归, 排序

📑 题目

实现快速排序算法，并说明其时间复杂度和优化方法。

💡 参考答案

快速排序（Quick Sort）：
- 分治算法，选取pivot将数组分为两部分
- 左边<pivot，右边>pivot
- 递归排序左右两部分

时间复杂度：
- 平均：O(n log n)
- 最坏：O(n²)（数组已有序时）
- 空间：O(log n)递归栈

优化方法：
1. 三数取中法选pivot，避免最坏情况
2. 小数组切换插入排序（减少递归开销）
3. 尾递归优化
4. 三路划分（处理大量重复元素）

💻 代码示例

// 标准快排 void quickSort(vector<int>& arr, int left, int right) { if (left >= right) return; // 三数取中法选pivot int mid = left + (right - left) / 2; if (arr[mid] < arr[left]) swap(arr[left], arr[mid]); if (arr[right] < arr[left]) swap(arr[left], arr[right]); if (arr[right] < arr[mid]) swap(arr[mid], arr[right]); int pivot = arr[mid]; swap(arr[mid], arr[right - 1]); // pivot放倒数第二 int i = left, j = right - 1; while (true) { while (arr[++i] < pivot); while (arr[--j] > pivot); if (i >= j) break; swap(arr[i], arr[j]); } swap(arr[i], arr[right - 1]); // pivot归位 quickSort(arr, left, i - 1); quickSort(arr, i + 1, right); } // 三路快排（处理重复元素） void quickSort3Way(vector<int>& arr, int left, int right) { if (left >= right) return; int pivot = arr[left]; int lt = left, gt = right, i = left + 1; while (i <= gt) { if (arr[i] < pivot) swap(arr[lt++], arr[i++]); else if (arr[i] > pivot) swap(arr[i], arr[gt--]); else i++; } // arr[left..lt-1] < pivot // arr[lt..gt] == pivot // arr[gt+1..right] > pivot quickSort3Way(arr, left, lt - 1); quickSort3Way(arr, gt + 1, right); }
