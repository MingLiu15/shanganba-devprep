**еҗҢжӯҘжңәеҲ¶пјҡ**

**жңҖдҪіе®һи·өпјҡ**

#include <iostream> #include <thread> #include <mutex> #include <condition_variable> #include <atomic> #include <future> #include <vector> std::mutex mtx; std::condition_variable cv; bool ready = false; // еҹәжң¬зәҝзЁӢеҮҪж•° void worker(int id) { std::unique_lock<std::mutex> lock(mtx); cv.wait(lock, []{ return ready; }); // зӯүеҫ…жқЎд»¶ std::cout << "Worker " << id << " is processing" << std::endl; } // еҺҹеӯҗеҸҳйҮҸпјҲж— й”Ғи®Ўж•°еҷЁпјү std::atomic<int> counter{0}; void increment() { for (int i = 0; i < 1000; ++i) { counter.fetch_add(1, std::memory_order_relaxed); } } // asyncејӮжӯҘд»»еҠЎ int compute(int x) { std::this_thread::sleep_for(std::chrono::seconds(1)); return x * x; } int main() { // 1. еҲӣе»әзәҝзЁӢ std::vector<std::thread> threads; for (int i = 0; i < 5; ++i) { threads.emplace_back(worker, i); } // 2. йҖҡзҹҘжүҖжңүзәҝзЁӢејҖе§Ӣ { std::lock_guard<std::mutex> lock(mtx); ready = true; } cv.notify_all(); // 3. зӯүеҫ…зәҝзЁӢе®ҢжҲҗ for (auto& t : threads) { t.join(); } // 4. еҺҹеӯҗж“ҚдҪңжөӢиҜ• std::thread t1(increment); std::thread t2(increment); t1.join(); t2.join(); std::cout << "Counter: " << counter << std::endl; // 2000 // 5. asyncејӮжӯҘи°ғз”Ё std::future<int> result = std::async(std::launch::async, compute, 5); std::cout << "Computing..." << std::endl; std::cout << "Result: " << result.get() << std::endl; // 25 // 6. жӯ»й”ҒйҒҝе…Қ std::mutex m1, m2; { std::lock(m1, m2); // еҗҢж—¶иҺ·еҸ–еӨҡдёӘй”Ғ std::lock_guard<std::mutex> lock1(m1, std::adopt_lock); std::lock_guard<std::mutex> lock2(m2, std::adopt_lock); // е®үе…ЁдҪҝз”ЁдёӨдёӘй”Ғ } return 0; }
